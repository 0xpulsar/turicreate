24 enum class combiner_filter { SRC, DST, ALL };

26 /**
27  * This is a reusable combiner object.
28  * It is expensive to create, so you should only make one of it for each
29  * combination of T and Combiner type. Essentially:
30  *
31  * \code
32  * combiner combiner(dc);
33  *
34  * combiner.perform_combine(graph, data, fn);
35  *
36  * \endcode
37  * Combiner must be a function of the form
38  * combiner(left, right) and mutates left with const right
39  */
40 template <typename T, typename Combiner>
41 class combiner {
42  public:
43  combiner(distributed_control& dc,
44  const Combiner& value_combiner)
45  : rmi(dc, this), value_combiner(value_combiner) { }
46 
47  void perform_combine(distributed_graph& g,
48  std::vector<T>& v,
49  combiner_filter filter=combiner_filter::ALL) {
50  ASSERT_EQ(v.size(), g.num_partitions());
51  master_locks.resize(v.size());
52  data = &v;
53  graph = &g;
54  rmi.barrier();
55  send_to_masters(filter);
56  send_to_children();
57  }
58 
59  void perform_sync(distributed_graph& g,
60  std::vector<T>& v) {
61  ASSERT_EQ(v.size(), g.num_partitions());
62  data = &v;
63  graph = &g;
64  rmi.barrier();
65  send_to_children();
66  }
67 
68  private:
69  void send_to_masters(combiner_filter filter) {
70  std::vector<size_t> my_parts;
71  if (filter == combiner_filter::SRC) {
72  my_parts = graph->my_src_vertex_partitions();
73  } else if (filter == combiner_filter::DST) {
74  my_parts = graph->my_dst_vertex_partitions();
75  } else {
76  my_parts = graph->my_vertex_partitions();
77  }
78 
79  parallel_for(0, my_parts.size(), [&](size_t idx) {
80  auto coord = my_parts[idx];
81  // send to master if I am not master
82  auto master = graph->get_partition_master(coord);
83  if (master != rmi.procid()) {
84  logstream(LOG_INFO) << "[Proc " << rmi.procid() << "] call to proc "
85  << master << " for partition " << coord
86  << " of size " << (*data)[coord].size()
87  << std::endl;
88  rmi.RPC_CALL(remote_call, combiner::receive_from_children)(master, coord, (*data)[coord]);
89  }
90  });
91  rmi.full_barrier();
92  }
93 
94  void send_to_children() {
95  for (size_t coord = 0; coord < graph->num_partitions(); ++coord) {
96  auto worker_list = graph->get_partition_workers(coord);
97  // if I am master for this partition and if there are children
98  if (graph->is_master_of_partition(coord) && worker_list.size() > 1) {
99  rmi.RPC_CALL(broadcast_call, combiner::receive_from_master)(
100  worker_list.begin() + 1, // skip the first worker, which is master
101  worker_list.end(),
102  coord,
103  (*data)[coord]);
104  }
105  }
106  rmi.full_barrier();
107  }
108 
109  void receive_from_children(size_t partition, const T& child_values) {
110  logstream(LOG_INFO) << "[Proc " << rmi.procid() << "] receive "
111  << "partition " << partition
112  << " of size " << child_values.size()
113  << std::endl;
114  // ASSERT_EQ(child_values.size(), data->at(partition).size());
115  ASSERT_LT(partition, data->size());
116  std::lock_guard<turi::mutex> guard(master_locks[partition]);
117  auto& my_values = (*data)[partition];
118  value_combiner(my_values, child_values);
119  }
120 
121  void receive_from_master(size_t partition, T& master_value) {
122  // logstream(LOG_INFO) << "Receiving Call for " << partition << std::endl;
123  // ASSERT_EQ(master_value.size(), data->at(partition).size());
124  ASSERT_LT(partition, data->size());
125  (*data)[partition] = std::move(master_value);
126  }
127 
128  dc_dist_object<combiner> rmi;
129  Combiner value_combiner;
130  distributed_graph* graph;
131  std::vector<T>* data;
132  std::vector<turi::mutex> master_locks;
133 };

160 template<typename VertexData, typename F>
161 inline void vertex_apply(distributed_graph& g, std::vector<VertexData>& vdata, F apply_func) {
162  for (auto partition_id: g.my_master_vertex_partitions()) {
163  apply_func(vdata[partition_id], partition_id);
164  }
165  auto dc = distributed_control::get_instance();
166  auto dummy_combine_fn = [](VertexData& a, const VertexData& b)->void {};
167  combiner<VertexData, decltype(dummy_combine_fn)> dummy_combiner(*dc, dummy_combine_fn);
168  dummy_combiner.perform_sync(g, vdata);
169 }

171 /**
172  * Create a vector of size_t num_partition, each of the element
173  * is created with BlockAllocator with the input of
174  * number of vertices in the local graph.
175  *
176  * Type T is the type of the vertex partition data to be allocated,
177  * it can be std::vector<flexible_type>, or EIGTODO Matrices
178  *
179  * BlockAllocator is a function that takes a size_t and return
180  * data of type T.
181  */
182 template <typename T, typename BlockAllocator>
183 std::vector<T>
184 create_partition_aligned_vertex_data(const distributed_graph& graph,
185  BlockAllocator block_allocator) {
186  std::vector<T> ret(graph.num_partitions());
187  for (auto coord: graph.my_vertex_partitions()) {
188  size_t partition_size = graph.num_vertices(coord);
189  ret[coord] = std::move(block_allocator(partition_size));
190  }
191  return ret;
192 }

194 /**
195  * Returns a vector of size num_partition, where vec[i] is the
196  * vertex data of that partition i, if current machine is the master
197  * of partition i. Otherwise, vec[i] is empty.
198  */
199 inline std::vector<std::vector<flexible_type>>
200 get_vertex_data_of_master_partitions(distributed_graph& graph, std::string field_name) {
201  std::vector<std::vector<flexible_type>> ret;
202  ret.resize(graph.num_partitions());
203  for (auto coord: graph.my_master_vertex_partitions()) {
204  std::vector<flexible_type> buffer;
205  auto vals = graph.local_graph().vertex_partition(coord).select_column(field_name);
206  size_t rows_read = vals->get_reader()->read_rows(0, vals->size(), buffer);
207  ret[coord] = buffer;
208  }
209  return ret;
210 }
