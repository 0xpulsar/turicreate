19 /**
20  * \ingroup unity
21  * The base class for all model objects. The model object is an object which
22  * conceptually contains a machine learning model, is implemented on the server,
23  * and exposed to the client via the cppipc system.
24  *
25  * The model object is designed to be highly generic, and exposes a simple
26  * interface which is abstractly that of a map from a string to a variant
27  * object (\ref variant_type). Implementations of the model must inherit from
28  * model_base and implement all functions.
29  *
30  * \ref simple_model is the most basic implementation of this model object.
31  *
32  * \see simple_model
33  */
34 class EXPORT model_base : public cppipc::ipc_object_base {
35  public:
36  typedef model_proxy proxy_object_type;
37 
38  virtual ~model_base() {}
39  /**
40  * Lists all the keys accessible in the map. All these keys are queryable
41  * via get_value.
42  */
43  virtual std::vector<std::string> list_keys() = 0;
44 
45  /**
46  * Returns the value of a particular key. To permit arbitrary queryability,
47  * an argument can be passed along with the query request. (For instance,
48  * the "key" may reference a vector of numbers, and the argument is used to
49  * identify the vector offset.). The returned object must be a copy/clone
50  * and should not contain pointers/references back into this model object.
51  * In other words, if the model object is destroyed, the returned object
52  * should not be invalidated.
53  */
54  virtual variant_type get_value(std::string key, variant_map_type& arg) = 0;
55 
56  inline virtual void save(oarchive& oarc) const {
57  oarc << get_version();
58  save_impl(oarc);
59  }
60 
61 
62  inline virtual void load(iarchive& iarc) {
63  size_t version = 0;
64  iarc >> version;
65  load_version(iarc, version);
66  }
67  /**
68  * Returns the name of the model.
69  */
70  virtual std::string name() = 0;
71 
72  /**
73  * Returns the current model version
74  */
75  virtual size_t get_version() const = 0;
76 
77  /**
78  * Serializes the model. Must save the model to the file format version
79  * matching that of get_version()
80  */
81  virtual void save_impl(oarchive& oarc) const = 0;
82 
83  /**
84  * Loads a model previously saved at a particular version number.
85  * Should raise an exception on failure.
86  */
87  virtual void load_version(iarchive& iarc, size_t version) = 0;
88 
89  REGISTRATION_BEGIN(model_base);
90  REGISTER(model_base::list_keys)
91  REGISTER(model_base::get_value)
92  REGISTER(model_base::name)
93  REGISTRATION_END
94 };
