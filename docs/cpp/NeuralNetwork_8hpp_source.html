28  // We won't do correctness checking here, that's for the validator.
29  std::unordered_set<std::string> layerOutputs;
30  for (const auto& output : spec.description().output()) {

33  return std::vector<std::string>(layerOutputs.begin(), layerOutputs.end());

36  // The classifier is a special case. Here, we need to not count as layer names the predicted
37  // feature name or predicted probabilities name. Additionally, we need to get the blob

40  inline std::vector<std::string> NeuralNetwork::outputNames<Specification::NeuralNetworkClassifier>(const CoreML::Specification::Model& spec, const Specification::NeuralNetworkClassifier& nnClassifier) {
41 
42  // We won't do correctness checking here, that's for the validator
43  std::unordered_set<std::string> layerOutputs;
44  for (const auto& output : spec.description().output()) {
45  if (output.name().compare(spec.description().predictedfeaturename()) != 0
46  && output.name().compare(spec.description().predictedprobabilitiesname()) != 0)
47  layerOutputs.insert(output.name());
48  }
49 
50  const std::string& probBlob = nnClassifier.labelprobabilitylayername();
51  if (probBlob.compare("") != 0) {
52  // then just add this to the set
53  layerOutputs.insert(probBlob);
54  }
55  else {
56  // for backwards compatibility with compiling old models (or poorly constructed
57  // ones)
58  std::unordered_set<std::string> allOutputLayerNames;
59  std::unordered_set<std::string> allInputLayerNames;
60  for (const auto& layer : nnClassifier.layers()) {
61  for (const auto& name : layer.input()) {
62  allInputLayerNames.insert(name);
63  }
64  for (const auto& name : layer.output()) {
65  allOutputLayerNames.insert(name);
66  }
67  }
68  for (const auto& name : allOutputLayerNames) {
69  if (allInputLayerNames.find(name) == allInputLayerNames.end()) {
70  layerOutputs.insert(name);
71  }
72  }
73  }
74  return std::vector<std::string>(layerOutputs.begin(), layerOutputs.end());
75 
76  }
