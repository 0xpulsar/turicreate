37 /**
38  *
39  * Solve a first_order_optimization_iterface model with an LBFGS
40  * implementation.
41  *
42  * The implementation is based on Algorithm 7.4 (pg 178) of [1].
43  *
44  * This subroutine solves an unconstrained minimization problem
45  * using the limited memory BFGS method. The routine is especially
46  * effective on problems involving a large number of variables. In
47  * a typical iteration of this method an approximation Hk to the
48  * inverse of the Hessian is obtained by applying M BFGS updates to
49  * a diagonal matrix Hk0, using information from the previous M steps.
50  * The user specifies the number M, which determines the amount of
51  * storage required by the routine. The user may also provide the
52  * diagonal matrices Hk0 if not satisfied with the default choice.
53  * The algorithm is described in [2].
54  *
55  * The user is required to calculate the function value and its
56  * gradient.
57  *
58  * The steplength is determined at each iteration by means of the
59  * line search routine MCVSRCH, which is a slight modification of
60  * the routine CSRCH written by More' and Thuente.
61  *
62  *
63  * References:
64  *
65  * (1) Wright S.J and J. Nocedal. Numerical optimization. Vol. 2.
66  * New York: Springer, 1999.
67  *
68  * (2) "On the limited memory BFGS method for large scale optimization", by D.
69  * Liu and J. Nocedal, Mathematical Programming B 45 (1989) 503-528.
70  *
71  * \param[in,out] model Model with first order optimization interface.
72  * \param[in] init_point Starting point for the solver.
73  * \param[in,out] opts Solver options.
74  * \param[in] reg Shared ptr to an interface to a smooth regularizer.
75  * \returns stats Solver return stats.
76  * \tparam Vector Sparse or dense gradient representation.
77  *
78 */
79 template <typename Vector = DenseVector>
80 inline solver_return lbfgs(first_order_opt_interface& model,
81  const DenseVector& init_point,
82  std::map<std::string, flexible_type>& opts,
83  const std::shared_ptr<smooth_regularizer_interface> reg=NULL){
84 
85  // Benchmarking utils.
86  timer t;
87  double start_time = t.current_time();
88  bool simple_mode = opts.count("simple_mode") && (opts.at("simple_mode"));
89 
90  if (!simple_mode) {
91  logprogress_stream << "Starting L-BFGS " << std::endl;
92  logprogress_stream
93  << "--------------------------------------------------------"
94  << std::endl;
95  std::stringstream ss;
96  ss.str("");
97  }
98 
99  // Step 1: Algorithm init
100  // ------------------------------------------------------------------------
101  // Check that all solver options are present.
102  size_t iter_limit = opts["max_iterations"];
103  double convergence_threshold = opts["convergence_threshold"];
104  double step_size = 1.0 / model.num_examples(); // Esimate of Lipshipz
105  int iters = 0;
106 
107  // Print status
108  auto header =
109  (simple_mode ? model.get_status_header({"Iteration", "Elapsed Time"})
110  : model.get_status_header({"Iteration", "Passes",
111  "Step size", "Elapsed Time"}));
112 
113  table_printer printer(header);
114  printer.print_header();
115 
116  int m = opts["lbfgs_memory_level"]; // Memory level in LBFGS
117  size_t n = model.num_variables(); // Dimension of point
118 
119  solver_return stats;
120 
121  // Keep track of previous point
122  DenseVector point = init_point;
123  DenseVector delta_point = point;
124  delta_point.zeros();
125 
126  // First compute the gradient. Sometimes, you already have the solution
127  // during the starting point. In these settings, you don't want to waste
128  // time performing the step.
129  Vector gradient(point.size());
130  DenseVector reg_gradient(point.size());
131  DenseVector delta_grad = gradient;
132  DenseVector new_grad = gradient;
133  delta_grad.zeros();
134 
135  // Add regularizer to gradients.
136  double func_value;
137  model.compute_first_order_statistics(point, gradient, func_value);
138  stats.num_passes++;
139  if (reg != NULL){
140  reg->compute_gradient(point, reg_gradient);
141  gradient += reg_gradient;
142  }
143  double residual = compute_residual(gradient);
144  double fprevious = func_value;
145  bool tune_step_size = true;
146 
147  // LBFGS storage
148  // The search steps and gradient differences are stored in a order
149  // controlled by the start point.
150  DenseMatrix y(n,m); // Step difference (prev m iters)
151  DenseMatrix s(n,m); // Gradient difference (prev m iters)
152  DenseVector H0(n); // Init Diagonal matrix (stores as a vector)
153  DenseVector q(n); // Storage required for the 2-loop recursion
154  DenseVector r(n); // Storage required for the 2-loop recursion
155  double beta; // Storage required for the 2-loop recursion
156  double gamma; // Scaling factor of Initial hessian
157  DenseVector rho(m); // Scaling factors (prev m iters)
158  DenseVector alpha(m); // Step sizes (prev m iters)
159 
160  int start_point = -1;
161  H0.ones();
162  s.zeros();
163  y.zeros();
164  r.zeros();
165  q.zeros();
166  rho.zeros();
167  alpha.zeros();
168  beta = 0;
169  gamma = 0;
170 
171 
172  // Nan Checking!
173  if (!std::isfinite(residual)) {
174  stats.status = OPTIMIZATION_STATUS::OPT_NUMERIC_OVERFLOW;
175  }
176 
177  // Step 2: Algorithm starts here
178  // ------------------------------------------------------------------------
179  // While not converged
180  ls_return ls_stats;
181  while((residual >= convergence_threshold) && (size_t(iters) < iter_limit)){
182 
183  // Perform gradient descent in the first iteration
184  if (iters == 0){
185 
186  // Line search
187  double reg_func = 0;
188  if (reg != NULL)
189  reg_func = reg->compute_function_value(point);
190 
191  ls_stats = more_thuente(model,
192  step_size,
193  func_value + reg_func,
194  point,
195  gradient,
196  -gradient,
197  reg);
198 
199  // Add info from line search
200  stats.func_evals += ls_stats.func_evals;
201  stats.gradient_evals += ls_stats.gradient_evals;
202  stats.num_passes += ls_stats.num_passes;
203 
204  // Line search failed
205  if (ls_stats.status == false){
206  stats.status = OPTIMIZATION_STATUS::OPT_LS_FAILURE;
207  break;
208  }
209 
210  delta_point = - ls_stats.step_size * gradient;
211  point = point + delta_point;
212 
213  // Two loop recursion to compute the direction
214  // Algorithm 7.4 of Reference [1]
215  }else{
216 
217  q = gradient;
218 
219  /**
220  * Data is stored in a cyclic format using the following indexiing:
221  *
222  * Iteration Storage location
223  * *****************************************************
224  * iter-1 start_point
225  * iter-2 (start_point + 1) % m
226  * ... ...
227  * iter-m (start_point + m - 1) % m
228  *
229  **/
230  for (int j = 0; j <= std::min(iters, m-1); j++){
231  int i = (start_point + j) % m;
232  alpha(i) = rho(i) * dot(s.col(i),q);
233  q = q - alpha(i) * y.col(i);
234  }
235 
236  // Scaling factor according to Pg 178 of [1]. This ensures that the
237  // problem is better scaled and that a step size of 1 is mostly accepted.
238  gamma = 1 / (squared_norm(y.col(start_point)) * rho(start_point));
239  r = gamma * (arma::diagmat(H0) * q);
240  for (int j = std::min(iters, m-1); j >=0 ; j--){
241  int i = (start_point + j) % m;
242  beta = rho(i) * dot(y.col(i), r);
243  r = r + s.col(i) * (alpha(i) - beta);
244  }
245 
246  // Update the new point and gradient
247  double reg_func = 0;
248  if (reg != NULL)
249  reg_func = reg->compute_function_value(point);
250 
251  if (tune_step_size == true){
252  ls_stats = more_thuente(model,
253  1,
254  func_value + reg_func,
255  point,
256  gradient,
257  -r,
258  reg);
259  // Add info from line search
260  stats.func_evals += ls_stats.func_evals;
261  stats.gradient_evals += ls_stats.gradient_evals;
262  stats.num_passes += ls_stats.num_passes;
263 
264  // Line search failed
265  if (ls_stats.status == false){
266  stats.status = OPTIMIZATION_STATUS::OPT_LS_FAILURE;
267  break;
268  }
269  tune_step_size = false;
270  }
271 
272  delta_point = - ls_stats.step_size * r;
273  point = point + delta_point;
274 
275  }
276 
277  // Numerical error: Insufficient progress.
278  if (arma::norm(delta_point, 2) <= OPTIMIZATION_ZERO){
279  stats.status = OPTIMIZATION_STATUS::OPT_NUMERIC_ERROR;
280  break;
281  }
282  // Numerical error: Numerical overflow. (Step size was too large)
283  if (!delta_point.is_finite()) {
284  stats.status = OPTIMIZATION_STATUS::OPT_NUMERIC_OVERFLOW;
285  break;
286  }
287 
288  // Compute the new gradient and store the difference in gradients
289  model.compute_first_order_statistics(point, new_grad, func_value);
290  stats.num_passes++;
291  if (reg != NULL){
292  reg->compute_gradient(point, reg_gradient);
293  new_grad += reg_gradient;
294  }
295  delta_grad = new_grad - gradient;
296  gradient = new_grad;
297  residual = compute_residual(gradient);
298 
299  // Check for nan's in the function value.
300  if(!std::isfinite(func_value)) {
301  stats.status = OPTIMIZATION_STATUS::OPT_NUMERIC_ERROR;
302  break;
303  }
304 
305 
306  if(func_value > fprevious){
307  tune_step_size = true;
308  }
309  fprevious = func_value;
310 
311 
312  // Store the previous gradient difference, step difference and rho
313  start_point = (start_point + 1) % m;
314  s.col(start_point) = delta_point;
315  y.col(start_point) = delta_grad;
316  rho(start_point) = 1/ (dot(delta_point, delta_grad));
317  iters++;
318 
319  // Log info for debugging.
320  logstream(LOG_INFO) << "Iters (" << iters << ") "
321  << "Passes (" << stats.num_passes << ") "
322  << "Residual (" << residual << ") "
323  << "Loss (" << func_value << ") "
324  << std::endl;
325 
326  // Print progress
327  auto stat_info =
328  (simple_mode
329  ? std::vector<std::string>{std::to_string(iters),
330  std::to_string(t.current_time())}
331  : std::vector<std::string>{std::to_string(iters),
332  std::to_string(stats.num_passes),
333  std::to_string(ls_stats.step_size),
334  std::to_string(t.current_time())});
335 
336  auto row = model.get_status(point, stat_info);
337  printer.print_progress_row_strs(iters, row);
338 
339  }
340  printer.print_footer();
341 
342 
343  // Step 3: Return optimization model status.
344  // ------------------------------------------------------------------------
345  if (stats.status == OPTIMIZATION_STATUS::OPT_UNSET) {
346  if (size_t(iters) < iter_limit){
347  stats.status = OPTIMIZATION_STATUS::OPT_OPTIMAL;
348  } else {
349  stats.status = OPTIMIZATION_STATUS::OPT_ITERATION_LIMIT;
350  }
351  }
352 
353  stats.iters = iters;
354  stats.residual = residual;
355  stats.func_value = func_value;
356  stats.solve_time = t.current_time() - start_time;
357  stats.solution = point;
358  stats.progress_table = printer.get_tracked_table();
359 
360  // Display solver stats
361  log_solver_summary_stats(stats, simple_mode);
362 
363  return stats;
364 }
