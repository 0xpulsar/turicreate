31 /**
32  * This is the object passed to the coroutine which allows the coroutine
33  * to read and write values. The expected usage pattern of the coroutine is:
34  *
35  * \code
36  * void fn(query_context& ctx) {
37  * while(1) {
38  * auto input = ctx.get_next(0); // from 1st input. get_next(1) for 2nd input, etc.
39  * auto output = get_output_buffer();
40  * // fill output buffer. Its just an sframe_rows
41  * ctx.emit(output); // only if output is non-empty.
42  * }
43  * }
44  * \endcode
45  */
46 class query_context {
47  public:
48  query_context();
49  query_context(const query_context&) = default;
50  query_context(query_context&&) = default;
51  ~query_context();
52  query_context(std::function<std::shared_ptr<sframe_rows>(size_t, bool)> callback_on_get_input,
53  std::function<emit_state(const std::shared_ptr<sframe_rows>&)> callback_on_emit,
54  size_t m_buffer_size,
55  emit_state initial_state);
56 
57  /**
58  * Requests for the next block for the given input.
59  */
60  std::shared_ptr<const sframe_rows> get_next(size_t input_number);
61 
62  /**
63  * Requests for the next block for the given input to the skipped.
64  */
65  void skip_next(size_t input_number);
66 
67  /**
68  * Returns a pointer to an output buffer.
69  */
70  std::shared_ptr<sframe_rows> get_output_buffer();
71 
72  /**
73  * Returns the initial state.
74  */
75  emit_state initial_state() const;
76 
77  /**
78  * Emits a collection of rows. The number of rows emitted
79  * MUST be the same as block_size(), except for the very last block
80  * of rows.
81  */
82  emit_state emit(const std::shared_ptr<sframe_rows>& rows);
83 
84  /**
85  * The commmunication block size.
86  */
87  inline size_t block_size() const {
88  return m_max_buffer_size;
89  }
90  private:

102 
103  emit_state m_initial_state;
