16 /**
17  * \ingroup fault
18  * Describes a wrapper around an array of zeromq messages part.
19  * Writes are performed through the insert_back() function.
20  * The standard pattern is:
21  * \code
22  * // creates a new message part and returning it
23  * zmq_msg_t* msg = msgvec.insert_back();
24  * zmq_msg_init_...(msg, ...)
25  * \endcode
26  *
27  * It can also be used for reading. It maintains a read index which
28  * is initialized to zero. Each call to read_next() will return the message
29  * part at the readindex, and increment the read index. If there are no more
30  * parts, the function will return NULL.
31  *
32  * The contents of the vector are automatically freed on destruction.
33  */
34 class EXPORT zmq_msg_vector {

40  }

42  clone_from(const_cast<zmq_msg_vector&>(other));
43  return *this;
44  }

50  void clone_from(zmq_msg_vector& other) {
51  clear();

54  zmq_msg_init(&msgs[i]);

56  }
57  }
