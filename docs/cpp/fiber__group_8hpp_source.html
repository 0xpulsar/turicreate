13 /**
14  * Defines a group of fibers. Analogous to the thread_group, but is meant
15  * to run only little user-mode threads. It is important that fibers never
16  * block, since there is no way to context switch out from a blocked fiber.
17  * The fiber_group uses the fiber_control singleton instance to manage its
18  * fibers.
19  * \ingroup threading
20  */
21 class fiber_group {
22  public:
23  typedef fiber_control::affinity_type affinity_type;
24 
25  private:
26  size_t stacksize;
27  affinity_type affinity;
28  atomic<size_t> threads_running;
29  mutex join_lock;
30  // to be triggered once the threads_running counter becomes 0
31  conditional join_cond;
32  // set to true if someone is waiting on a join()
33  bool join_waiting;
34 
35  bool exception_raised;
36  std::string exception_value;
37  mutex exception_lock;
38 
39  inline void increment_running_counter() {
40  threads_running.inc();
41  }
42 
43  inline void decrement_running_counter() {
44  // now, a bit of care is needed here
45  size_t r = threads_running.dec();
46  if (r == 0) {
47  join_lock.lock();
48  if (join_waiting) {
49  join_cond.signal();
50  }
51  join_lock.unlock();
52  }
53  }

57  fiber_group* group);
