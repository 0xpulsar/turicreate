34  * Stores the intermediate or complete result of a histogram stream.
35  *
36  * Attributes
37  * ----------
38  * bins : flex_list<flex_int>
39  * The counts in each bin, in the range scale_min to scale_max.
40  * min : flexible_type
41  * The lowest value encountered in the source SArray.
42  * max : flexible_type
43  * The highest value encountered in the source SArray.
44  * scale_min : double
45  * The low end of the range represented by the bins.
46  * scale_max : double
47  * The high end of the range represented by the bins.
48  *
49  * Methods
50  * -------
51  * init(flexible_type value1, flexible_type value2)
52  * Initialize the result with the range represented by two values.
53  * init(flexible_type value1, flexible_type value2, double scale1, double scale2)
54  * Initialize the result with the range represented by two values and
55  * the scale range (should be >= value range) represented by two scale
56  * values.
57  * rescale(double new_min, double new_max)
58  * Rescale the result to the range represented by the provided values.
59  */

91 /*
92  * histogram()
93  *
94  * Implements Optimal Streaming Histogram (sort-of) as described in
95  * http://blog.amplitude.com/2014/08/06/optimal-streaming-histograms/.
96  * dtype of sarray can be flex_int or flex_float
97  * histogram always gives bins as flex_ints (bin counts are positive integers).
98  *
99  * Attributes
100  * ----------
101  *
102  * Methods
103  * ----------
104  * init(const gl_sarray& source)
105  * Initialize the histogram with an SArray as input.
106  * eof()
107  * Returns true if the streaming histogram has covered all input, false
108  * otherwise.
109  * get()
110  * Bins (up to) the next BATCH_SIZE values from the input, and returns
111  * the histogram (result type, as shown below) representing the current
112  * distribution of values seen so far.
113  */
114 typedef transformation<gl_sarray, histogram_result> histogram_parent;
115 class histogram : public histogram_parent {
116  public:
117  virtual std::vector<histogram_result> split_input(size_t num_threads) override;
118  virtual void merge_results(std::vector<histogram_result>& transformers) override;
119  virtual void init(const gl_sarray& source, size_t batch_size) override;
120 };
