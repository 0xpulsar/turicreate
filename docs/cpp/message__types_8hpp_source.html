16 /**
17  * \internal
18  * \ingroup fault
19  * The internal structure of a message for an object
20  * This struct is filled in using direct pointers to the
21  * actual message string. Thus the member pointers should not
22  * be freed.
23  */
24 struct query_object_message {
25  struct header_type {
26  /// Flags identifying message properties.
27  /// for instance, if this is a query message or an update message
28  uint64_t flags;
29  /**
30  * Each update message has a a ID identifying the query.
31  * This should be generated randomly by the sender.
32  * This ID does not require strong randomness guarantees, but is
33  * used to only to identify unprocessed messages in the event
34  * of a machine failure.
35  */
36  uint64_t msgid;
37  } header;
38  /// The message contents
39  char* msg;
40  /// The length of the message.
41  size_t msglen;
42 
43  /// Parses the message header.
44  /// No copies are made, and msg will point directly into the zeromq
45  /// message structure.
46  void parse(zmq_msg_vector& data);
47 
48  /// Generates the zeromq message corresponding to this message
49  /// Gives away to zeromq the pointer to the message contents
50  /// Only appends are performed to outdata.
51  void write(zmq_msg_vector& outdata);
52 };
