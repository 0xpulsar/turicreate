20 /**
21  * Represents a rectangular area within an image.
22  *
23  * The coordinate system is defined by the user. Any rect without a positive
24  * width and a positive height is an empty or null rect.
25  */
26 struct image_box {
27  image_box() = default;
28  image_box(float x, float y, float width, float height)
29  : x(x), y(y), width(width), height(height)
30  {}
31 
32  // Computes the area if the width and height are positive, otherwise returns 0
33  float area() const {
34  if (width < 0.f || height < 0.f) return 0.f;
35  return width * height;
36  }
37 
38  // Divides each coordinate and length by the appropriate normalizer.
39  void normalize(float image_width, float image_height);
40 
41  // Sets this instance to the intersection with the given image_box. If no
42  // intersection exists, then the result will have area() of 0.f (and may have
43  // negative width or height).
44  void clip(image_box clip_box = image_box(0.f, 0.f, 1.f, 1.f));
45 
46  float x = 0.f;
47  float y = 0.f;
48  float width = 0.f;
49  float height = 0.f;
50 };
