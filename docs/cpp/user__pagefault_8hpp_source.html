24 /**
25  * This implements a user mode page fault handler.
26  *
27  * The basic mechanics of operation are not very complicated.
28  * We first install a segfault_handler.
29  *
30  * When you ask for some memory, we use mmap to allocate a region, but set
31  * memory protection on it to PROT_NONE (disable both read and writes to the
32  * region). This way, every memory access to the region will trigger a segfault.
33  *
34  * When the memory is accessed, the segfault handler (segv_handler()) is
35  * triggered, and try to fill in the data in the page.
36  * To do so, (fill_pages()) we set the protection on the page to
37  * PROT_READ | PROT_WRITE (enable read + write), call a callback function to
38  * fill in the data, then sets the protection on the page to PROT_READ. Then we
39  * return from the segfault handler which allows the program to resume
40  * execution correctly (but cannot write to the memory, only read).
41  *
42  * We keep a queue of pages we have committed (add_to_access_queue()). and when
43  * the queue size becomes too large, we start evicting/decommitting pages
44  * (handle_eviction()). Essentially we are doing FIFO caching. To "evict"
45  * a set of pages simply involves calling madvise(MADV_DONTNEED) on the pages.
46  *
47  * A bunch of additional maintenance stuff are needed:
48  * - a queue of committed pages (used to manage eviction)
49  * - The set of all regions (page_sets) allocated, sorted by address
50  * to permit fast binary searches). this is used to figure out the mapping
51  * between address and callback.
52  *
53  * One key design consideration is the careful use of mmap. The way the
54  * procedure is defined above only calls mmap once for the entire region.
55  * decommitting and committing memory is done by use of memory protection
56  * and madvise. Alternate implementation built entirely around mmap is possible
57  * where we use completely unallocated memory addresses, and use mmap to map
58  * the pages in as required. But this requires a large number of calls to mmap,
59  * which uses up a lot of kernel datastructures, which will cause mmap to fail
60  * with ENOMEM after a while.
61  *
62  * Also, with regards to parallelism, the kernel API does not quite provide
63  * sufficient capability to handle parallel accesses correctly. What is missing
64  * is an atomic "fill and enable page read/write" function. Essentially, while
65  * the callback function is filling in the pages for a particular memory address,
66  * some other thread can read from the same memory addresses and get erroneous
67  * values. mremap could in theory be used to enable this by having the callback
68  * fill to alternate pages, and using mremap to atomically swap those pages in.
69  * However, see the design consideration above regarding excessive use of mmap.
70  * Also, mremap is not available on OS X.
71  */

90 /**
91  * The structure used to maintain all the metadata about the pagefault handler.

93 struct userpf_page_set {
94  char* begin = nullptr; ///< The start of the managed address
95  char* end = nullptr; ///< The end of the manged address
96  size_t length = 0; ///< end - start
97  /**
98  * The number of "pages". Each pagefault will trigger the fill of one
99  * "page". The size of the page is predefined at compile time.
100  * \ref PAGE_SIZE.
101  */
102  size_t num_large_pages = 0;
103 
104  bool writable = false;
105 
106  /**
107  * Internal datastructures. Should not touch.
108  */
109  /// one lock for each page to handle at least, a limited amount of parallelism
110  std::vector<simple_spinlock> locks;
111 
112  /// one bit for each page that is resident (has physical pages associated)
113  dense_bitset resident;
114  /// one bit for each page that is dirty (written to but not flushed)
115  dense_bitset dirty;
116  /**
117  * one bit for each page that is maintained by the pagefault handler's
118  * pagefile instead of using the callback. Basically these pages
119  * are pages which have been written to, and then evicted.
120  */
121  dense_bitset pagefile_maintained;
122 
123  /**
124  * the callback handler which is called to fill in new page
125  */
126  userpf_handler_callback fill_callback;
127  /**
128  * the callback handler which is called when the memory is freed.
129  * (this allows an external procedure to potentially persist the data)
130  * This must be reentrant with the filling callback handler since the release
131  * callback may read the entire memory region (which required filling)
132  */
133  userpf_release_callback release_callback;
134 
135  /**
136  * pagefile_allocation[i] is the handle to the pagefile storing the disk
137  * backed memory. (size_t)(-1) if not allocated.
138  */
139  std::vector<size_t> pagefile_allocations;

184 /**
185  * Allocates a block of memory of a certain length, where the contents of
186  * the memory are to be filled using the specified callback function.
187  *
188  * Returns a page_set pointer where page_set->begin is the memory address
189  * at which on-demand paging will be performed. The callback is triggered
190  * to fill in the pages on demand.
191  *
192  * The pagefault handler is an std::function object of the form:
193  * \code
194  * size_t fill_callback(userpf_page_set* ps,
195  * char* address,
196  * size_t fill_length) {
197  * // MUST fill in the contents of address to address + fill_length
198  * // most not write out of the bounds of [address, address+fill_length)
199  * // for instance, to fill in the array such that
200  * // arr = (size_t*)(ps->begin)
201  * // arr[i] == i
202  * // we do the following:
203  *
204  * size_t* root = (size_t*) ps->begin;
205  * size_t* s_addr = (size_t*) page_address;
206  *
207  * size_t begin_value = s_addr - root;
208  * size_t num_to_fill = minimum_fill_length / sizeof(size_t);
209  *
210  * for (size_t i = 0; i < num_to_fill; ++i) {
211  * s_addr[i] = i + begin_value;
212  * }
213  * }
214  * \endcode
215  *
216  * The release_callback is an std::function object of the form:
217  * \code
218  * size_t release_callback(userpf_page_set* ps) {
219  * // for instance, we may write out the contents of
220  * // ps->begin to ps->end to disk here.
221  * }
222  * \endcode
223  * The release_callback function *must* be reentrant with the fill_callback
224  * function. This is because memory accesses performed by release_callback into
225  * to the pointer may trigger the fill_callback to fill in the memory regions.
226  *
227  * If writable is set (default false), the user_pagefault handler will perform
228  * automatic read/write/eviction.
229  *
230  * \param length The number of bytes to allocate
231  * \param fill_callback The callback to be triggered when a page range needs
232  * to be filled.
233  * \param release_callback The callback to be triggered the the memory is
234  * released, i.e. release(pf) is called.
235  * \param writable If the page range is to be writable. The pagefault handler
236  * will handle paging.
237  * The returned object MUST NOT be freed or deallocated. use \ref release.
238  */
239 userpf_page_set* allocate(size_t length,
240  userpf_handler_callback fill_callback,
241  userpf_release_callback release_callback = nullptr,
242  bool writable = true);
