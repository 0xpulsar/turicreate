20  /**
21  * \ingroup util
22  * \brief Implements a blocking queue useful for producer/consumer models.
23  * Similar to the \ref blocking_queue, but requires all threads waiting on
24  * the queue to be fibers. Also only supports the basic wait on dequeue
25  * operation, and not the other timed_wait, or wait_until_empty operations
26  * supported by the \ref blocking_queue
27  */
28  template<typename T>
29  class fiber_blocking_queue {
30  protected:
31 
32  typedef typename std::deque<T> queue_type;
33 
34  bool m_alive;
35  queue_type m_queue;
36  mutex m_mutex;
37  std::queue<size_t> fiber_queue;
38 
39  volatile uint16_t sleeping;
40 
41 
42  public:

44  //! creates a blocking queue
45  fiber_blocking_queue() : m_alive(true),sleeping(0) { }

47  void wake_a_fiber() {
48  if (!fiber_queue.empty()) {
49  size_t fiber_id = fiber_queue.front();
50  fiber_queue.pop();
51  fiber_control::schedule_tid(fiber_id);
52  }
53  }
54 
55  void wake_all_fibers() {
56  while(!fiber_queue.empty()) {
57  size_t fiber_id = fiber_queue.front();
58  fiber_queue.pop();
59  fiber_control::schedule_tid(fiber_id);
60  }
61  }
62 
63  void fiber_sleep() {
64  fiber_queue.push(fiber_control::get_tid());
65  fiber_control::deschedule_self(&m_mutex.m_mut);
66  m_mutex.lock();
67  }

69  //! Add an element to the blocking queue
70  inline void enqueue(const T& elem, bool wake_consumer = true) {
71  m_mutex.lock();
72  m_queue.push_back(elem);
73  // Signal threads waiting on the queue
74  if (wake_consumer && sleeping) wake_a_fiber();
75  m_mutex.unlock();
76  }
77 
78  //! Add an element to the blocking queue
79  inline void enqueue_to_head(const T& elem) {
80  m_mutex.lock();
81  m_queue.push_front(elem);
82  // Signal threads waiting on the queue
83  if (sleeping) wake_a_fiber();
84  m_mutex.unlock();
85  }

103  inline bool wait_for_data() {
104  m_mutex.lock();
105  bool success = false;
106  // Wait while the queue is empty and this queue is alive
107  while(m_queue.empty() && m_alive) {
108  sleeping++;
109  fiber_sleep();
110  sleeping--;
111  }
112  // An element has been added or a signal was raised
113  if(!m_queue.empty()) {
114  success = true;
115  }
116  m_mutex.unlock();
117 
118  return success;
119  }

122  /**
123  * Blocks until an element is available in the queue
124  * or until stop_blocking() is called.
125  * The return value is a pair of <T value, bool success>
126  * If "success" if set, then "value" is valid and
127  * is an element popped from the queue.
128  * If "success" is false, stop_blocking() was called
129  * and the queue has been destroyed.
130  */
131  inline std::pair<T, bool> dequeue() {
132 
133  m_mutex.lock();
134  T elem = T();
135  bool success = false;
136  // Wait while the queue is empty and this queue is alive
137  while(m_queue.empty() && m_alive) {
138  sleeping++;
139  fiber_sleep();
140  sleeping--;
141  }
142  // An element has been added or a signal was raised
143  if(!m_queue.empty()) {
144  success = true;
145  elem = m_queue.front();
146  m_queue.pop_front();
147  }
148  m_mutex.unlock();
149 
150  return std::make_pair(elem, success);
151  }
152 
153  /**
154  * Returns an element if the queue has an entry.
155  * returns [item, false] otherwise.
156  */
157  inline std::pair<T, bool> try_dequeue() {
158  if (m_queue.empty() || m_alive == false) return std::make_pair(T(), false);
159  m_mutex.lock();
160  T elem = T();
161  // Wait while the queue is empty and this queue is alive
162  if (m_queue.empty() || m_alive == false) {
163  m_mutex.unlock();
164  return std::make_pair(elem, false);
165  }
166  else {
167  elem = m_queue.front();
168  m_queue.pop_front();
169  }
170  m_mutex.unlock();
171 
172  return std::make_pair(elem, true);
173  }
174 
175  //! Returns true if the queue is empty
176  inline bool empty() {
177  m_mutex.lock();
178  bool res = m_queue.empty();
179  m_mutex.unlock();
180  return res;
181  }
182 
183  /** Wakes up all threads waiting on the queue whether
184  or not an element is available. Once this function is called,
185  all existing and future dequeue operations will return with failure.
186  Note that there could be elements remaining in the queue after
187  stop_blocking() is called.
188  */
189  inline void stop_blocking() {
190  m_mutex.lock();
191  m_alive = false;
192  wake_all_fibers();
193  m_mutex.unlock();
194  }
195 
196  /**
197  Resumes operation of the fiber_blocking_queue. Future calls to
198  dequeue will proceed as normal.
199  */
200  inline void start_blocking() {
201  m_mutex.lock();
202  m_alive = true;
203  m_mutex.unlock();
204  }
205 
206  //! get the current size of the queue
207  inline size_t size() {
208  return m_queue.size();
209  }
210 
211 
212  /**
213  * Causes any threads currently blocking on a dequeue to wake up
214  * and evaluate the state of the queue. If the queue is empty,
215  * the threads will return back to sleep immediately. If the queue
216  * is destroyed through stop_blocking, all threads will return.
217  */
218  void broadcast() {
219  m_mutex.lock();
220  wake_all_fibers();
221  m_mutex.unlock();
222  }
223 
224 
225  ~fiber_blocking_queue() {
226  m_alive = false;
227  broadcast();
228  }
229  }; // end of fiber_blocking_queue class
