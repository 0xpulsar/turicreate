52  /// Wait on the barrier until numthreads has called wait
53  inline void wait() const {
54  if (!alive) return;
55  mutex.lock();
56  // set waiting;
57  fiber_handles[called] = fiber_control::get_tid();
58  called++;
59  bool listening_on = barrier_sense;
60  if (called == needed) {
61  // if I have reached the required limit, wait up. Set waiting
62  // to 0 to make sure everyone wakes up
63  std::vector<size_t> to_wake = fiber_handles;
64  called = 0;
65  barrier_release = barrier_sense;
66  barrier_sense = !barrier_sense;
67  // clear all waiting, wake everyone. (less 1 since current thread
68  // is already awake)
69  for (size_t i = 0;i < to_wake.size() - 1; ++i) {
70  fiber_control::schedule_tid(to_wake[i]);
71  }
72  } else {
73  // while no one has broadcasted, sleep
74  while(barrier_release != listening_on && alive) {
75  fiber_control::deschedule_self(&mutex.m_mut);
76  mutex.lock();
77  }
78  }
79  mutex.unlock();
80  }
81  }; // end of conditional
