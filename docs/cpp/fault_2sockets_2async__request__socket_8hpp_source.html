33 /**
34  * \ingroup fault
35  * Constructs a zookeeper backed asynchronous request socket.
36  * Will automatically retry sockets.
37  * This object is multi-threaded. Calls can be made from any thread.
38  *
39  * \note
40  * The design is to use DEALER sockets to communicate with the REP sockets
41  * on the other side. However, background sends are an issue because zeromq
42  * does not let me do sends and receives in seperate threads.
43  * The solution is to create 3 sockets.
44  * - A DEALER socket to talk to the REP socket remotely.
45  * This can then be attached to the pollset and callbacks will then set the
46  * future.
47  * - An inproc PULL socket also attached to the pollset. Messages received
48  * on the PULL socket are forwarded out the DEALER socket.
49  * - An inproc PUSH socket to send stuff to the PULL socket.
50  *
51  * An issue is the retries are somewhat tricky to manage. So unlike the
52  * synchronous request_socket, this socket does not do retries. Any failure
53  * will return a status in the future, and it is up to the client to perform
54  * the retry.
55  */
56 class EXPORT async_request_socket {

58  typedef libfault::message_reply message_reply;

60  typedef libfault::future_reply future_reply;

68  * \param masterkey The master object key where requests (via request_master)
69  * are sent.
70  * \param slavekeys The slave object keys where requests (via request_any)
71  * are sent.
72  *
73  * If keyval is NULL, masterkey and slavekeys must directly refer to ZeroMQ

75  * messaging wrapper around ZeroMQ which provides asynchronous messaging

97  * as well as a status which may contain an error. The message_reply pointer
98  * must be freed by the user when done.

100  * Possible errors are:
101  *
102  * - \b EHOSTUNREACH Target machine is currently unreachable. This could be

198  // finds a promise by ID, returns and unregisters the promise
199  inline boost::promise<message_reply*>* get_promise(uint64_t promise_id) {
200  boost::unordered_map<uint64_t, boost::promise<message_reply*>* >
201  ::iterator iter = promises.find(promise_id);
202  if (iter == promises.end()) return NULL;
203  else {
204  boost::promise<message_reply*>* ret = iter->second;
205  promises.erase(iter);
206  return ret;
