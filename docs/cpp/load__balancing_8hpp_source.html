15  /**
16  * Given a binary m by n matrix M, representing n jobs
17  * being processed by m processers, and job j can only be
18  * processed by i's where M[i,j] == 1
19  *
20  * The solver *approximately
21  * solves" the following optimization problem:
22  *
23  * Let M' be an m by n solution matrix:
24  * - M'.col(j) has one and only one non-zero entry. // each job processed
25  * once and only once
26  * - If M'[i, j] == 1, then M[i, j] == 1 // processor i is in the constraint
27  * set
28  * - L(i) = sum(M'.row(i)) is the load of i
29  * - Minimize the largest load: max_i L(i)
30  */
31 
32  /**
33  * Heuristics:
34  * First we try to assign job j to machine i == j % numprocs, if i is in the constraint set.
35  * Then for the remanining partitions, we assign randomly.
36  *
37  * Returns {assignemnt[job_id], max_load}
38  */
39  inline std::pair<std::vector<size_t>, double> solve_genearlized_load_balancing(
40  const EIGTODO::SparseMatrix<size_t>& constraint_mat) {
41  size_t nprocs = constraint_mat.n_rows;
42  size_t njobs = constraint_mat.n_cols;
43 
44  std::vector<size_t> assignment(njobs, (size_t)(-1));
45  std::vector<size_t> load(nprocs, 0);
46  size_t max_load;
47 
48  // Assign j to j % proc if feasible.
49  for (size_t j = 0; j < njobs; ++j) {
50  size_t i = j % nprocs;
51  if (constraint_mat.coeff(i,j) == 1) {
52  assignment[j] = i;
53  load[i]++;
54  break;
55  }
56  }
57 
58  // Assign j to hash(j) % constraint_set(j)
59  for (size_t j = 0; j < njobs; ++j) {
60  if (assignment[j] == (size_t)(-1)) {
61  size_t nnz = constraint_mat.col(j).nonZeros();
62  size_t rand = turi::hash64(j) % nnz;
63  for (size_t i = 0; i < nprocs; ++i) {
64  if (constraint_mat.coeff(i, j) == 1) {
65  if (rand == 0) {
66  assignment[j] = i;
67  load[i]++;
68  break;
69  }
70  rand--;
71  }
72  }
73  }
74  }
75 
76  max_load = *std::max_element(load.begin(), load.end());
77 
78  // Sanity check
79 #ifndef NDEBUG
80  for (size_t j = 0; j < assignment.size(); ++j) {
81  DASSERT_LT(assignment[j], nprocs);
82  size_t val = constraint_mat.coeff(assignment[j], j);
83  DASSERT_EQ(val, 1);
84  }
85 #endif
86  return {assignment, (double)max_load};
87  }
