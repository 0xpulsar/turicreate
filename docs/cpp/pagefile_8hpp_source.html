91  private:
92  /**
93  * This structure maintains the arena of size "arena_size". i.e.
94  * It only maintains allocations of *exactly* that size.
95  *
96  * Essentially I am interpreting the file as a collection of N consecutive
97  * sections of arena_size bytes. A bitfield of length N is used to denote whether
98  * a particular section is in use or not.
99  */
100  struct arena {
101  turi::mutex pagefile_lock;
102  int pagefile_handle = -1;
103  size_t current_pagefile_length = 0;
104  size_t arena_size = 0;
105  /**
106  * bitfield of which positions within the arena are being used
107  * and which are not. if allocations.get(i) is false, the section is not
108  * being used.
109  */
110  dense_bitset allocations;
111  };
112 
113  /** An array of all the arenas we have. These must be sorted by arena_size
114  * in increasing order.
115  */
116  arena m_arenas[NUM_ARENAS];
117 
118  /// The number of arenas
119  size_t m_num_arenas = 0;
120 
121  /// The size of the largest arena
122  size_t m_max_arena_size = 0;
123 
124  /**
125  * The total number of bytes allocated
126  */
127  turi::atomic<size_t> m_total_allocated_bytes;
128 
129  /**
130  * Total number of alloc calls
131  */
132  turi::atomic<size_t> m_num_allocations_made;
133 
134  /**
135  * Every allocation (accessed by the \ref allocate(), \ref read()
136  * \ref write() and \ref free() function), references a handle to one of
137  * these datastructures that tell us what is the size that was allocated,
138  * and its current location (which arena it is in, whether it is compressed,
139  * etc).
140  */
141  struct allocation{
142  size_t stored_size = 0;
143  size_t prelz4_size = 0;
144  size_t original_size = 0;
145  std::pair<size_t, size_t> arena_number_and_offset = {-1, -1};
146  bool compressed = false;
147  simple_spinlock allocation_lock;
148  };
149  /**
150  * Since the user just sees integer handles, this is a map from integer
151  * handles to the allocation metadata.
152  */
153  turi::mutex m_lock;
154  std::unordered_map<size_t, std::shared_ptr<allocation>> m_handle_to_allocation;
155  size_t m_next_handle_id;
156 
157  /*
158  * These functions below here handle the lower level with-in arena allocations.
159  * Each arena allocation is referenced by an "arena_number_and_offset"
160  * address.
161  * Basically, the arena number is the index into the m_arenas array; i.e.
162  * which arena. The offset is the position within the pagefile.
163  * i.e. to get to the data, you seek to to the offset arena_size * offset
164  * within the pagefile.
165  */
166 
167  /**
168  * Given a size, find the best fit arena
169  */
170  size_t best_fit_arena(size_t size);
171 
172  /**
173  * allocates an arena range.
174  * Return a pair of arena number and offset
175  */
176  std::pair<size_t, size_t> allocate_arena(size_t size);
177 
178  /**
179  * deallocates an arena from a number and offset pair.
180  */
181  void deallocate_arena(std::pair<size_t, size_t> arena_number_and_offset);
182  /**
183  * Reads the data stored at the arena address arena_number_and_offset
184  */
185  void read_arena(std::pair<size_t, size_t> arena_number_and_offset, char* data, size_t len);
186  void write_arena(std::pair<size_t, size_t> arena_number_and_offset, char* data, size_t len);
187 
188 };
